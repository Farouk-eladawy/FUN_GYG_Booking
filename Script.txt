    async function sendLogoutAlert(reason = 'login_page_detected') {
        const payload = {
            system: 'GYG Bookings',
            status: 'logged_out',
            reason,
            url: window.location.href,
            host: window.location.host,

                        logStep(`📣 Logout alert sent (${res.status})`, res.status >= 200 && res.status < 300 ? 'INFO' : 'WARNING');
                        if (res.status >= 200 && res.status < 300) markLogoutAlertSent();
                    },
                    onerror: function() {
                        logStep('❌ Logout alert failed to send', 'ERROR');
                    }
                });
            } else {
                const resp = await fetch(WEBHOOK_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload),
                    keepalive: true,
                });
                logStep(`📣 Logout alert sent (${resp.status})`, resp.ok ? 'INFO' : 'WARNING');
                if (resp.ok) markLogoutAlertSent();

    function setStorage(key, value) {
        try {
            if (typeof GM_setValue !== 'undefined') {

            } else {
                localStorage.setItem(key, JSON.stringify(value));

        } catch (e) {

        }
    }

    /**
     * استرجاع البيانات من التخزين المحلي أو GM storage
     */
    function getStorage(key, defaultValue = null) {
        try {
            if (typeof GM_getValue !== 'undefined') {
                return GM_getValue(key, defaultValue);
            } else {
                const stored = localStorage.getItem(key);
                return stored ? JSON.parse(stored) : defaultValue;
            }
        } catch (e) {
            console.warn('[Session Keeper] Storage read failed:', e);
            return defaultValue;
        }
    }

    /**
     * إرسال ping للحفاظ على الجلسة
     */
    async function sendSessionPing() {
        const pingUrl = pickPingUrl();
        const now = Date.now();

        try {
            // استخدام fetch مع إعدادات keep-alive
            const response = await fetch(pingUrl, {
                method: 'GET',
                credentials: 'include',
                keepalive: true,
                cache: 'no-cache',
                headers: {
                    'X-Requested-With': 'XMLHttpRequest',
                    'X-Session-Keeper': 'true'
                }
            });

            lastPingTime = now;
            setStorage(SESSION_KEEPER_CONFIG.STORAGE_KEYS.LAST_PING, now);

            if (response.ok) {
                logStep(`🔄 Session ping successful (${response.status}) - ${pingUrl}`, "NETWORK");
                return true;
            } else {
                if (response.status === 401 && shouldSendLogoutAlert()) {
                    await sendLogoutAlert('ping_unauthorized');
                }
                logStep(`⚠️ Session ping returned ${response.status} - ${pingUrl}`, "WARNING");
                return false;
            }

        } catch (error) {
            // Fallback إلى GM_xmlhttpRequest إذا فشل fetch
            return new Promise((resolve) => {
                if (typeof GM_xmlhttpRequest !== 'undefined') {
                    GM_xmlhttpRequest({
                        method: 'GET',
                        url: pingUrl,
                        headers: {
                            'X-Requested-With': 'XMLHttpRequest',
                            'X-Session-Keeper': 'true'
                        },
                        onload: function(response) {
                            lastPingTime = now;
                            setStorage(SESSION_KEEPER_CONFIG.STORAGE_KEYS.LAST_PING, now);

                            if (response.status >= 200 && response.status < 400) {
                                logStep(`🔄 Session ping successful (GM) (${response.status}) - ${pingUrl}`, "NETWORK");
                                resolve(true);
                            } else {
                                if (response.status === 401 && shouldSendLogoutAlert()) {
                                    sendLogoutAlert('gm_ping_unauthorized');
                                }
                                logStep(`⚠️ Session ping failed (GM) (${response.status}) - ${pingUrl}`, "WARNING");
                                resolve(false);
                            }
                        },
                        onerror: function() {
                            logStep(`❌ Session ping error (GM) - ${pingUrl}`, "ERROR");
                            resolve(false);
                        }
                    });
                } else {
                    logStep(`❌ Session ping failed - ${error.message}`, "ERROR");
                    resolve(false);
                }
            });
        }
    }

    /**
     * فحص ما إذا كان المستخدم في صفحة تسجيل الدخول

    function isOnLoginPage() {
        try {
            const u = new URL(window.location.href);
            const host = u.host.toLowerCase();
            const path = u.pathname.toLowerCase().replace(/\/+$/, '');
            const explicit = host === 'supplier.getyourguide.com' && path === '/auth/login';
            const strictStarts = host === 'supplier.getyourguide.com' && path.startsWith('/auth/login');
            const heuristics = ['/login', '/signin', '/auth', '/authenticate'].some(ind => path.includes(ind));
            return explicit || strictStarts || heuristics;
        } catch (e) {
            const href = window.location.href.toLowerCase();
            return href.includes('supplier.getyourguide.com/auth/login') ||
                   ['/login', '/signin', '/auth', '/authenticate'].some(indicator => href.includes(indicator));
        }
    }

    /**

    function saveCurrentPage() {

    /**
     * استعادة الصفحة المحفوظة بعد تسجيل الدخول
     */
    function restoreSavedPage() {
        const savedPage = getStorage(SESSION_KEEPER_CONFIG.STORAGE_KEYS.SAVED_PAGE);
        if (savedPage && savedPage !== window.location.href) {
            logStep(`🔄 Restoring saved page: ${savedPage}`, "INFO");
            setTimeout(() => {
                window.location.href = savedPage;
            }, 2000);
            return true;
        }
        return false;
    }

    /**
     * تسجيل النشاط وإعادة تعيين المؤقت
     */
    function recordActivity() {
        lastActivityTime = Date.now();
        setStorage(SESSION_KEEPER_CONFIG.STORAGE_KEYS.LAST_ACTIVITY, lastActivityTime);
    }

    /**
     * إضافة مستمعي الأحداث للنشاط
     */
    function setupActivityListeners() {
        SESSION_KEEPER_CONFIG.ACTIVITY_RESET_EVENTS.forEach(eventType => {
            const listener = () => recordActivity();
            document.addEventListener(eventType, listener, { passive: true });
            activityListeners.push({ type: eventType, listener });
        });
        logStep(`👂 Activity listeners setup for: ${SESSION_KEEPER_CONFIG.ACTIVITY_RESET_EVENTS.join(', ')}`, "DEBUG");
    }

    /**
     * إزالة مستمعي الأحداث
     */
    function removeActivityListeners() {
        activityListeners.forEach(({ type, listener }) => {
            document.removeEventListener(type, listener);
        });
        activityListeners = [];
    }

    /**
     * فحص ما إذا كان يجب إعادة تحميل الجلسة
     */
    function shouldRefreshSession() {
        const now = Date.now();
        const timeSinceLastPing = now - lastPingTime;
        const timeSinceLastActivity = now - lastActivityTime;

        // إعادة التحميل إذا لم يكن هناك ping ناجح لفترة طويلة
        // ولكن فقط إذا لم تكن المعالجة متوقفة
        return (
            timeSinceLastPing > SESSION_KEEPER_CONFIG.REFRESH_THRESHOLD &&
            timeSinceLastActivity < SESSION_KEEPER_CONFIG.REFRESH_THRESHOLD &&
            !isProcessingPaused
        );
    }

    /**
     * تنفيذ دورة Session Keeper
     */
    async function sessionKeeperCycle() {
        try {
            // فحص إذا كنا في صفحة تسجيل الدخول
            if (isOnLoginPage()) {
                logStep("🔐 Login page detected, attempting to restore saved page", "WARNING");
                if (shouldSendLogoutAlert()) {
                    await sendLogoutAlert('login_page_detected');
                }
                if (!restoreSavedPage()) {
                    // إذا لم تكن هناك صفحة محفوظة، احفظ الصفحة الرئيسية
                    setStorage(SESSION_KEEPER_CONFIG.STORAGE_KEYS.SAVED_PAGE,
                              'https://supplier.getyourguide.com/bookings');
                }
                return;
            }

            // حفظ الصفحة الحالية
            saveCurrentPage();

            // إرسال ping
            const pingSuccess = await sendSessionPing();

            // فحص ما إذا كان يجب إعادة التحميل
            if (shouldRefreshSession()) {
                logStep("🔄 Session refresh threshold reached, reloading page", "WARNING");
                setTimeout(() => {
                    if (!isProcessingPaused) {
                        window.location.reload();
                    }
                }, 1000);
                return;
            }

            // تسجيل حالة الجلسة
            const timeSinceLastPing = Date.now() - lastPingTime;
            const timeSinceLastActivity = Date.now() - lastActivityTime;

            logStep(`💓 Session heartbeat - Ping: ${Math.round(timeSinceLastPing/1000/60)}m ago, Activity: ${Math.round(timeSinceLastActivity/1000/60)}m ago`, "DEBUG");

        } catch (error) {
            logStep(`❌ Session keeper cycle error: ${error.message}`, "ERROR");
        }
    }

    /**
     * بدء نظام Session Keeper
     */
    function startSessionKeeper() {
        // تهيئة الأوقات
        const now = Date.now();
        lastPingTime = getStorage(SESSION_KEEPER_CONFIG.STORAGE_KEYS.LAST_PING, now - SESSION_KEEPER_CONFIG.PING_INTERVAL);
        lastActivityTime = getStorage(SESSION_KEEPER_CONFIG.STORAGE_KEYS.LAST_ACTIVITY, now);

        // تسجيل بداية الجلسة
        setStorage(SESSION_KEEPER_CONFIG.STORAGE_KEYS.SESSION_START, now);

        // إعداد مستمعي النشاط
        setupActivityListeners();

        // بدء الدورة الدورية
        sessionKeeperInterval = setInterval(sessionKeeperCycle, SESSION_KEEPER_CONFIG.PING_INTERVAL);

        // تنفيذ دورة أولى فورية
        setTimeout(sessionKeeperCycle, 2000);

        logStep("🚀 Session Keeper started - Keep-alive pings every 4 minutes", "SUCCESS");
        logStep(`📊 Session config - Ping interval: ${SESSION_KEEPER_CONFIG.PING_INTERVAL/1000/60}m, Refresh threshold: ${SESSION_KEEPER_CONFIG.REFRESH_THRESHOLD/1000/60}m`, "DEBUG");
    }

    /**
     * إيقاف نظام Session Keeper

        if (sessionKeeperInterval) {
            clearInterval(sessionKeeperInterval);
            sessionKeeperInterval = null;
        }
        removeActivityListeners();
        logStep("🛑 Session Keeper stopped", "INFO");

    /**
     * جعل عنصر قابل للسحب
     */
    function makeDraggable(element, handle) {
        let pos1 = 0, pos2 = 0, pos3 = 0, pos4 = 0;

        handle.onmousedown = dragMouseDown;

        function dragMouseDown(e) {
            e = e || window.event;
            e.preventDefault();
            pos3 = e.clientX;
            pos4 = e.clientY;
            document.onmouseup = closeDragElement;
            document.onmousemove = elementDrag;
        }

        function elementDrag(e) {
            e = e || window.event;
            e.preventDefault();
            pos1 = pos3 - e.clientX;
            pos2 = pos4 - e.clientY;
            pos3 = e.clientX;
            pos4 = e.clientY;

            // الحفاظ على اللوحة ضمن حدود النافذة
            const newTop = (element.offsetTop - pos2);
            const newLeft = (element.offsetLeft - pos1);

            if (newTop > 0 && newTop < (window.innerHeight - 50)) {
                element.style.top = newTop + "px";
            }

            if (newLeft > 0 && newLeft < (window.innerWidth - 100)) {
                element.style.left = newLeft + "px";
            }
        }

        function closeDragElement() {
            document.onmouseup = null;
            document.onmousemove = null;
        }
    }

    /**
     * جعل العنصر قابل لتغيير الحجم
     */
    function makeResizable(element, handle) {
        let startX, startY, startWidth, startHeight;

        handle.addEventListener('mousedown', function(e) {
            e.preventDefault();

            // تمييز اللوحة أثناء تغيير الحجم
            element.classList.add('resizing');

            startX = e.clientX;
            startY = e.clientY;
            startWidth = parseInt(document.defaultView.getComputedStyle(element).width, 10);
            startHeight = parseInt(document.defaultView.getComputedStyle(element).height, 10);

            document.documentElement.addEventListener('mousemove', resize, false);
            document.documentElement.addEventListener('mouseup', stopResize, false);
        }, false);

        function resize(e) {
            const newWidth = Math.max(300, startWidth + e.clientX - startX); // الحد الأدنى للعرض 300 بكسل
            const newHeight = Math.max(200, startHeight + e.clientY - startY); // الحد الأدنى للارتفاع 200 بكسل

            // تطبيق الحجم الجديد مع مراعاة حدود النافذة
            const maxWidth = window.innerWidth * 0.95;
            const maxHeight = window.innerHeight * 0.9;

            element.style.width = Math.min(newWidth, maxWidth) + 'px';
            element.style.height = Math.min(newHeight, maxHeight) + 'px';
        }

        function stopResize() {
            element.classList.remove('resizing');
            document.documentElement.removeEventListener('mousemove', resize, false);
            document.documentElement.removeEventListener('mouseup', stopResize, false);
        }
    }

    /**
     * تنظيف الذاكرة المحلية من البيانات القديمة
     */
    function clearOldCache() {
        const cache = getBookingsCache();
        const now = Date.now();
        const initialCount = Object.keys(cache).length;

        let removed = 0;
        for (const ref in cache) {
            if (now - cache[ref].timestamp > CACHE_EXPIRY) {
                delete cache[ref];
                removed++;
            }
        }

        if (removed > 0) {
            saveBookingsCache(cache);
            logStep(`Cache cleaned: removed ${removed} old bookings`, 'INFO');
        }

        return removed;
    }

    /**
     * تنسيق التاريخ
     */
    function formatDate(dateString) {
        if (!dateString) return null;
        const months = {January:"01",February:"02",March:"03",April:"04",May:"05",June:"06",July:"07",August:"08",September:"09",October:"10",November:"11",December:"12"};
        dateString = dateString.replace(/(\d+)(st|nd|rd|th)/gi, '$1');
        const match = dateString.match(/([A-Za-z]+),?\s+([A-Za-z]+)\s+(\d{1,2}),\s+(\d{4})(?:\s+(\d{1,2}:\d{2}\s*[APMapm]{2}))?/);
        if (match) {
            const [, , month, day, year, time] = match;
            const formatted = `${months[month]}/${day.padStart(2, '0')}/${year}`;
            return time ? `${formatted} ${formatTime(time.trim())}` : formatted;
        }
        return dateString;
    }

    /**
     * تنسيق الوقت
     */
    function formatTime(str) {
        const m = str.match(/(\d{1,2}):(\d{2})\s*([APMapm]{2})/);
        if (m) return `${m[1].padStart(2, '0')}:${m[2]} ${m[3].toUpperCase()}`;
        return str;
    }

    /**
     * الحصول على تاريخ منسّق للفلاتر
     */
    function getFormattedDate() {
        const d = new Date(), next = new Date();
        next.setDate(d.getDate() + 10);
        const f = d => d.toISOString().split('T')[0];
        return { today: f(d), nextDate: f(next) };
    }

    /**
     * مسح وتنظيف النص
     */
    function cleanText(text) {
        if (!text) return '';
        return text.trim().replace(/\s+/g, ' ');
    }

    /**
     * تحويل قيمة نصية إلى رقم
     */
    function extractNumberValue(text) {
        if (!text) return 0;
        const match = text.match(/(\d+(\.\d+)?)/);
        return match ? parseFloat(match[1]) : 0;
    }

    /**
     * استخراج وتحليل الإضافات (Add-Ons) باستخدام الذكاء الاصطناعي
     */
    function extractAndAnalyzeAddOns(booking) {
        // استخراج الإضافات الأساسية من الحجز
        const rawAddOns = [...booking.querySelectorAll('[data-testid="participants-breakdown-items"] li')]
            .map(el => el.textContent.trim())
            .filter(t => !/(Adult|Child|Infant|Student|Youth)/i.test(t));

        if (rawAddOns.length === 0) {
            return {
                formatted: "",
                original: [],
                categories: {},
                value: 0
            };
        }

        // تصنيف الإضافات حسب النوع باستخدام قواعد ذكية
        const categories = {
            transportation: [],
            meals: [],
            activities: [],
            equipment: [],
            special: [],
            other: []
        };

        let totalValue = 0;

        // تحليل كل إضافة
        rawAddOns.forEach(addon => {
            // استخراج السعر إن وجد
            const priceMatch = addon.match(/€(\d+(\.\d+)?)/);
            const price = priceMatch ? parseFloat(priceMatch[1]) : 0;
            totalValue += price;

            const lowerAddon = addon.toLowerCase();

            // تصنيف ذكي باستخدام الكلمات المفتاحية
            if (/transfer|transport|pickup|car|bus|taxi/i.test(lowerAddon)) {
                categories.transportation.push(addon);
            } else if (/lunch|dinner|breakfast|meal|food|drink|beverage/i.test(lowerAddon)) {
                categories.meals.push(addon);
            } else if (/tour|activity|visit|entrance|ticket|show/i.test(lowerAddon)) {
                categories.activities.push(addon);
            } else if (/equipment|gear|rental|mask|suit|board/i.test(lowerAddon)) {
                categories.equipment.push(addon);
            } else if (/vip|private|exclusive|custom|special/i.test(lowerAddon)) {
                categories.special.push(addon);
            } else {
                categories.other.push(addon);
            }
        });

        // تنظيف الفئات الفارغة
        Object.keys(categories).forEach(key => {
            if (categories[key].length === 0) {
                delete categories[key];
            }
        });

        // تنسيق النص بطريقة ذكية
        let formattedAddOns = "";

        // إنشاء تنسيق مفيد ومنظم
        if (Object.keys(categories).length > 0) {
            const parts = [];

            // ترتيب الإضافات حسب الأهمية والقيمة
            if (categories.special && categories.special.length) {
                parts.push(`Special: ${categories.special.join(", ")}`);
            }
            if (categories.activities && categories.activities.length) {
                parts.push(`Activities: ${categories.activities.join(", ")}`);
            }
            if (categories.transportation && categories.transportation.length) {
                parts.push(`Transport: ${categories.transportation.join(", ")}`);
            }
            if (categories.meals && categories.meals.length) {
                parts.push(`Meals: ${categories.meals.join(", ")}`);
            }
            if (categories.equipment && categories.equipment.length) {
                parts.push(`Equipment: ${categories.equipment.join(", ")}`);
            }
            if (categories.other && categories.other.length) {
                parts.push(`Other: ${categories.other.join(", ")}`);
            }

            formattedAddOns = parts.join(" | ");
        } else {
            formattedAddOns = rawAddOns.join(", ");
        }

        return {
            formatted: formattedAddOns,
            original: rawAddOns,
            categories: categories,
            value: totalValue
        };
    }

    /**
     * التواصل مع واجهة برمجة DeepSek للذكاء الاصطناعي
     */
    async function analyzeWithDeepSek(data, analysisType) {
        return new Promise((resolve, reject) => {
            GM_xmlhttpRequest({
                method: "POST",
                url: DEEPSEK_ENDPOINT,
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': `Bearer ${DEEPSEK_API_KEY}`
                },
                data: JSON.stringify({
                    type: analysisType,
                    data: data
                }),
                onload: function(response) {
                    if (response.status >= 200 && response.status < 300) {
                        try {
                            resolve(JSON.parse(response.responseText));
                        } catch (e) {
                            reject(new Error('Failed to parse DeepSek response'));
                        }
                    } else {
                        reject(new Error(`DeepSek API responded with status: ${response.status}`));
                    }
                },
                onerror: function(error) {
                    reject(new Error(`DeepSek connection failed: ${error.statusText || 'unknown error'}`));
                }
            });
        }).catch(error => {
            logStep(`DeepSek analysis failed: ${error.message}`, 'ERROR');
            return null;
        });
    }

    /**
     * تحليل أنماط الحجوزات باستخدام الذكاء الاصطناعي
     */
    function analyzeBookingPatterns(bookings) {
        // إحصائيات وأنماط
        const stats = {
            totalBookings: bookings.length,
            newBookings: 0,
            canceledBookings: 0,
            destinationCounts: {},
            productCounts: {},
            customerCountries: {},
            addOnsData: {
                totalBookingsWithAddOns: 0,
                popularCategories: {},
                averageAddOnValue: 0,
                totalAddOnValue: 0
            },
            timeDistribution: {
                morning: 0,   // 6-12
                afternoon: 0, // 12-18
                evening: 0    // 18-24
            }
        };

        bookings.forEach(booking => {
            // حساب أنواع الحجوزات
            if (booking.changeType === 'new') {
                stats.newBookings++;
            } else if (booking.changeType === 'canceled') {
                stats.canceledBookings++;
            }

            // تجميع حسب الوجهة
            const destination = booking["des"] || "غير محدد";
            stats.destinationCounts[destination] = (stats.destinationCounts[destination] || 0) + 1;

            // تجميع حسب المنتج
            const product = booking["trip Name"] || "غير محدد";
            stats.productCounts[product] = (stats.productCounts[product] || 0) + 1;

            // تجميع حسب بلد العميل
            const country = booking["Customer Country"] || "غير محدد";
            stats.customerCountries[country] = (stats.customerCountries[country] || 0) + 1;

            // تحليل بيانات الإضافات
            if (booking["Add-Ons Raw"] && booking["Add-Ons Raw"].length > 0) {
                stats.addOnsData.totalBookingsWithAddOns++;

                // جمع قيمة الإضافات
                if (booking["Add-Ons Value"]) {
                    const valueMatch = booking["Add-Ons Value"].match(/€(\d+(\.\d+)?)/);
                    if (valueMatch) {
                        const addOnValue = parseFloat(valueMatch[1]);
                        stats.addOnsData.totalAddOnValue += addOnValue;
                    }
                }

                // تحليل فئات الإضافات
                if (booking["Add - Ons"]) {
                    const categories = booking["Add - Ons"].split(" | ");
                    categories.forEach(cat => {
                        const categoryMatch = cat.match(/^([^:]+):/);
                        if (categoryMatch) {
                            const category = categoryMatch[1].trim();
                            stats.addOnsData.popularCategories[category] = (stats.addOnsData.popularCategories[category] || 0) + 1;
                        }
                    });
                }
            }

            // تحليل توزيع الوقت
            if (booking["Date Trip"]) {
                try {
                    const timeMatch = booking["Date Trip"].match(/(\d{1,2}):(\d{2})\s*([APMapm]{2})/);
                    if (timeMatch) {
                        let hour = parseInt(timeMatch[1]);
                        const period = timeMatch[3].toLowerCase();

                        if (period === 'pm' && hour < 12) {
                            hour += 12;
                        } else if (period === 'am' && hour === 12) {
                            hour = 0;
                        }

                        if (hour >= 6 && hour < 12) {
                            stats.timeDistribution.morning++;
                        } else if (hour >= 12 && hour < 18) {
                            stats.timeDistribution.afternoon++;
                        } else {
                            stats.timeDistribution.evening++;
                        }
                    }
                } catch (e) {
                    // تجاهل أخطاء تحليل الوقت
                }
            }
        });

        // حساب متوسط قيمة الإضافات
        if (stats.addOnsData.totalBookingsWithAddOns > 0) {
            stats.addOnsData.averageAddOnValue = stats.addOnsData.totalAddOnValue / stats.addOnsData.totalBookingsWithAddOns;
        }

        // تحليل وتقديم نصائح ذكية
        const insights = generateInsights(stats);

        // سجل الإحصائيات
        logStep('Analyzed booking patterns with AI', 'AI', stats);

        // إضافة النصائح إلى لوحة اللوج
        addAIInsight(insights);

        return stats;
    }

    /**
     * توليد نصائح ذكية محسّنة
     */
    function generateInsights(stats) {
        let insights = [];

        // تحليل اتجاهات الوجهات
        if (Object.keys(stats.destinationCounts).length > 0) {
            const topDestination = Object.entries(stats.destinationCounts)
                .sort((a, b) => b[1] - a[1])[0];
            insights.push(`Top destination: ${topDestination[0]} (${topDestination[1]} bookings)`);
        }

        // تحليل المنتجات الأكثر طلبًا
        if (Object.keys(stats.productCounts).length > 0) {
            const topProduct = Object.entries(stats.productCounts)
                .sort((a, b) => b[1] - a[1])[0];
            insights.push(`Top tour: ${topProduct[0]} (${topProduct[1]} bookings)`);
        }

        // تحليل توزيع الوقت
        const timeDistribution = stats.timeDistribution;
        const totalTimeBookings = timeDistribution.morning + timeDistribution.afternoon + timeDistribution.evening;
        if (totalTimeBookings > 0) {
            const peakTime = Object.entries(timeDistribution)
                .sort((a, b) => b[1] - a[1])[0];

            const timeLabels = {
                morning: "Morning (6-12)",
                afternoon: "Afternoon (12-18)",
                evening: "Evening (18-24)"
            };

            insights.push(`Peak booking time: ${timeLabels[peakTime[0]]} (${Math.round((peakTime[1] / totalTimeBookings) * 100)}%)`);
        }

        // نسبة الإلغاء
        if (stats.totalBookings > 0 && stats.canceledBookings > 0) {
            const cancelRate = (stats.canceledBookings / stats.totalBookings) * 100;
            if (cancelRate > 20) {
                insights.push(`High cancellation rate: ${Math.round(cancelRate)}% - Consider reviewing policies`);
            } else if (cancelRate < 5) {
                insights.push(`Low cancellation rate: ${Math.round(cancelRate)}% - Excellent performance!`);
            } else {
                insights.push(`Cancellation rate: ${Math.round(cancelRate)}%`);
            }
        }

        // تحليل الإضافات
        if (stats.addOnsData.totalBookingsWithAddOns > 0) {
            insights.push(`${stats.addOnsData.totalBookingsWithAddOns} bookings (${Math.round((stats.addOnsData.totalBookingsWithAddOns / stats.totalBookings) * 100)}%) include add-ons`);

            // متوسط قيمة الإضافات
            if (stats.addOnsData.averageAddOnValue > 0) {
                insights.push(`Average add-on value: €${stats.addOnsData.averageAddOnValue.toFixed(2)}`);
            }

            // الفئات الأكثر شعبية
            if (Object.keys(stats.addOnsData.popularCategories).length > 0) {
                const topCategory = Object.entries(stats.addOnsData.popularCategories)
                    .sort((a, b) => b[1] - a[1])[0];
                insights.push(`Most popular add-on: ${topCategory[0]} (${topCategory[1]} times)`);
            }
        }

        return insights.join(' | ');
    }

    /**
     * تحليل ودراسة التغييرات المكتشفة
     */
    function analyzeChanges(pendingChanges) {
        if (pendingChanges.length === 0) return;

        // تصنيف التغييرات
        const changeTypes = {
            new: pendingChanges.filter(b => b.changeType === 'new'),
            canceled: pendingChanges.filter(b => b.changeType === 'canceled'),
            updated: pendingChanges.filter(b => b.changeType === 'updated')
        };

        // تحديث إحصائيات اللوج
        logStats.total += pendingChanges.length;
        logStats.new += changeTypes.new.length;
        logStats.canceled += changeTypes.canceled.length;
        logStats.updated += changeTypes.updated.length;
        updateLogStats();

        // سجل ملخص التغييرات
        logStep(`Detected ${pendingChanges.length} changes: ${changeTypes.new.length} new, ${changeTypes.canceled.length} canceled, ${changeTypes.updated.length} updated`, 'PROCESS');

        // تحليل أنماط الحجوزات إذا كانت هناك تغييرات كافية
        if (pendingChanges.length > 3) {
            analyzeBookingPatterns(pendingChanges);
        }

        return changeTypes;
    }

                result[key] += parseInt(m[1]);
            }
        });

        return result;
    }

    /**
     * محاولة فتح قسم Show breakdown واستخراج القيم (مُدمج من AI Booking 2.txt)
     */
    async function expandBreakdownAndExtract(booking) {
        try {
            const btns = Array.from(booking.querySelectorAll('button, a'));
            const breakdownBtn = btns.find(b => (b.textContent || '').trim().toLowerCase().includes('show breakdown'));
            if (breakdownBtn) {
                breakdownBtn.click();
                await new Promise(r => setTimeout(r, 600));
            }

            const text = (booking.innerText || '').replace(/\s+/g, ' ');
            // أولاً: نحاول التقاط القيم من سطر "Total" مباشرة لضمان الدقة
            // شكل السطر يكون عادةً: "Total: €21.00 €14.70 30.00%"
            const totalLineMatch = text.match(/Total[^€]*€\s*([0-9.,]+)[^€]*€\s*([0-9.,]+)/i);
            const retailMatch = text.match(/Retail\s*Price[^€]*€\s*([0-9.,]+)/i);
            const revenueMatch = text.match(/Revenue[^€]*€\s*([0-9.,]+)/i);

            // أنماط متعددة لالتقاط نسبة العمولة لتحقيق أعلى توافق
            const supplierCommissionMatch = text.match(/Supplier\s+commission\s+rate\s*:\s*([0-9]+(?:\.[0-9]+)?)\s*%/i);
            const totalCommissionMatch = text.match(/Total\s+commission\s+rate[^%]*([0-9]+(?:\.[0-9]+)?)\s*%/i);

            const headerIndex = text.toLowerCase().indexOf('commission breakdown');
            let percentAfterHeader = null;
            if (headerIndex !== -1) {
                const slice = text.slice(headerIndex, Math.min(headerIndex + 500, text.length));
                const m = slice.match(/([0-9]+(?:\.[0-9]+)?)\s*%/);
                if (m) percentAfterHeader = parseFloat(m[1]);
            }

            const allPercents = Array.from(text.matchAll(/([0-9]+(?:\.[0-9]+)?)\s*%/g)).map(m => parseFloat(m[1]));
            const fallbackPercent = allPercents.length ? allPercents[allPercents.length - 1] : null;

            // إذا وُجد سطر الإجمالي، نستخدمه كمرجع أساسي
            const retail = totalLineMatch
                ? `€${totalLineMatch[1].replace(/,/g, '')}`
                : (retailMatch ? `€${retailMatch[1].replace(/,/g, '')}` : null);
            const revenue = totalLineMatch
                ? `€${totalLineMatch[2].replace(/,/g, '')}`
                : (revenueMatch ? `€${revenueMatch[1].replace(/,/g, '')}` : null);
            const percentValue = supplierCommissionMatch ? parseFloat(supplierCommissionMatch[1])
                                 : (totalCommissionMatch ? parseFloat(totalCommissionMatch[1])
                                 : (percentAfterHeader != null ? percentAfterHeader
                                 : (fallbackPercent != null ? fallbackPercent : null)));
            // أيرتابل (نوع Percent) يتوقع قيمة كسريّة بين 0 و 1
            const commission = percentValue != null ? (percentValue / 100) : null;
            return { retail, revenue, commission };
        } catch {
            return { retail: null, revenue: null, commission: null };
        }
    }

    /**
     * ===================================
     * وظائف اكتشاف التغييرات ومقارنة البيانات
     * ===================================
     */

    /**
     * المقارنة الذكية بين الحجوزات واكتشاف التغييرات
     */
    function detectChanges(bookings) {
        const cache = getBookingsCache();
        const pendingChanges = [];
        logStep(`Starting intelligent change detection for ${bookings.length} bookings`, 'PROCESS');

        let totalFound = 0;
        let unchangedCount = 0;

        bookings.forEach(booking => {
            const ref = booking["Booking Nr."];
            if (!ref) return; // تجاهل الحجوزات بدون مرجع

            const cachedBooking = cache[ref];

            // إذا لم يكن محفوظًا من قبل، فهو جديد
            if (!cachedBooking) {
                totalFound++;
                booking.changeType = 'new';
                booking.changes = ['New booking created'];
                booking.approved = AUTO_SEND_NEW_BOOKINGS; // موافقة تلقائية حسب الإعدادات
                pendingChanges.push(booking);

                logStep(`New booking detected: ${ref}`, 'BOOKING');
                addBookingPreview(booking);
                return;
            }

            // اكتشاف التغييرات بين الحجز الحالي والمخزن
            const changes = [];
            let statusChanged = false;
            let hasImportantChanges = false;

            for (const key in booking) {
                // تجاهل بعض الحقول التي لا تحتاج للمقارنة
                if (key === 'changeType' || key === 'changes' || key === 'timestamp' || key === 'approved') continue;

                // تنظيف وإعداد القيم للمقارنة
                let cachedValue = cachedBooking[key] !== undefined ? cachedBooking[key] : '';
                let newValue = booking[key] !== undefined ? booking[key] : '';

                // المقارنة الذكية للقيم
                if (JSON.stringify(newValue) !== JSON.stringify(cachedValue)) {
                    // تحقق إذا كان هذا تغيير في حالة الحجز
                    if (key === "Booking Status" && newValue === "Canceled") {
                        statusChanged = true;
                        changes.push(`⚠️ ${key}: Booking canceled (from "${cachedValue}" to "${newValue}")`);
                        hasImportantChanges = true;
                    }
                    // تغيير في التفاصيل الهامة للحجز
                    else if (['Date Trip', 'trip Name', 'Customer Name', 'Customer Email'].includes(key)) {
                        changes.push(`🔔 ${key}: from "${cachedValue}" to "${newValue}"`);
                        hasImportantChanges = true;
                    }
                    // تغييرات أخرى أقل أهمية
                    else {
                        changes.push(`${key}: from "${cachedValue}" to "${newValue}"`);
                    }
                }
            }

            // إضافة إلى قائمة التغييرات إذا تم اكتشاف تغييرات
            if (changes.length > 0) {
                totalFound++;
                booking.changeType = statusChanged ? 'canceled' : 'updated';
                booking.changes = changes;

                // الحجوزات الملغاة تحصل على موافقة تلقائية مثل الحجوزات الجديدة
                booking.approved = (statusChanged && AUTO_SEND_CANCELED_BOOKINGS) ||
                                  (booking.changeType === 'updated' && AUTO_SEND_UPDATES);

                pendingChanges.push(booking);

                // سجل التغييرات للمراجعة
                logStep(`${changes.length} changes detected in booking ${ref} (${booking.changeType})`, hasImportantChanges ? 'WARNING' : 'COMPARE');
                addBookingPreview(booking, changes);
            } else {
                unchangedCount++;
            }
        });

        logStats.unchanged += unchangedCount;

        logStep(`Change detection completed: ${totalFound} bookings with changes, ${unchangedCount} unchanged`, 'PROCESS');
        return pendingChanges;
    }

    /**
     * مقارنة حقل واحد بين الحجز القديم والجديد
     */
    function compareField(oldValue, newValue, fieldName) {
        // تنظيف وتحويل القيم للمقارنة
        oldValue = oldValue !== undefined ? oldValue : '';
        newValue = newValue !== undefined ? newValue : '';

        // حالات خاصة لبعض الحقول
        if (fieldName === "Date Trip") {
            // مقارنة التواريخ
            const oldDate = new Date(oldValue);
            const newDate = new Date(newValue);
            if (!isNaN(oldDate.getTime()) && !isNaN(newDate.getTime())) {
                return oldDate.getTime() === newDate.getTime();
            }
        }

        // مقارنة عامة
        return JSON.stringify(oldValue) === JSON.stringify(newValue);
    }

    /**
     * دمج البيانات القديمة والجديدة مع الحفاظ على البيانات القديمة
     * تم تحسينها لتطبيق نهج القائمة البيضاء
     */
    function mergeBookingData(oldData, newData) {
        const result = {};

        // نستخدم فقط الحقول المسموح بها
        ALLOWED_AIRTABLE_FIELDS.forEach(field => {
            // نستخدم القيمة الجديدة إذا كانت موجودة، وإلا نستخدم القيمة القديمة
            if (newData[field] !== undefined) {
                result[field] = newData[field];
            } else if (oldData[field] !== undefined) {
                result[field] = oldData[field];
            }
        });

        return result;
    }

    /**
     * ===================================
     * وظائف الاتصال بـ Airtable
     * ===================================
     */

    /**
     * الحصول على معرف السجل في Airtable
     */
    function getBookingRecordId(ref) {
        return new Promise((resolve, reject) => {
            GM_xmlhttpRequest({
                method: "GET",
                url: `https://api.airtable.com/v0/${AIRTABLE_BASE_ID}/${AIRTABLE_TABLE_NAME}?filterByFormula={Booking Nr.}='${ref}'`,
                headers: { Authorization: `Bearer ${AIRTABLE_API_KEY}` },
                onload: res => {
                    try {
                        const data = JSON.parse(res.responseText);
                        resolve(data.records?.[0]?.id || null);
                    } catch (e) { reject(e); }
                },
                onerror: reject
            });
        });
    }

    /**
     * الحصول على قيمة حقل "pickup time" من Airtable عبر مرجع الحجز
     */
    function fetchAirtablePickupTime(ref) {
        return new Promise((resolve, reject) => {
            try {
                GM_xmlhttpRequest({
                    method: "GET",
                    url: `https://api.airtable.com/v0/${AIRTABLE_BASE_ID}/${encodeURIComponent(AIRTABLE_TABLE_NAME)}?filterByFormula=${encodeURIComponent(`{Booking Nr.}='${ref}'`)}`,
                    headers: { Authorization: `Bearer ${AIRTABLE_API_KEY}` },
                    onload: res => {
                        try {
                            const data = JSON.parse(res.responseText);
                            const fields = (data && data.records && data.records[0] && data.records[0].fields) || null;
                            const pickupRaw = fields && (fields["pickup time"] || fields["Pickup time"] || fields["Pickup Time"]);
                            const pickup = typeof pickupRaw === 'string' ? pickupRaw : (pickupRaw && pickupRaw.name) ? pickupRaw.name : null;
                            resolve(pickup || null);
                        } catch (e) {
                            reject(e);
                        }
                    },
                    onerror: err => reject(err)
                });
            } catch (err) {
                reject(err);
            }
        });
    }

    /**
     * فتح نافذة Specify time للحجز وتمرير وقت الالتقاط وإرساله للعميل
     * يُعاد:
     *  - true: تم الإرسال بنجاح
     *  - false: فشلت محاولة الإرسال (يجب الإيقاف وعدم الانتقال)
     *  - null: لا توجد قيمة pickup time في Airtable (لا يتم الإيقاف)
     */
    async function autoSpecifyPickupTime(bookingElement, ref) {
        try {
            // البحث عن زر Specify time داخل بطاقة الحجز
            const specifyBtn = bookingElement.querySelector('[data-testid="edit-pickup-time"]')
                || Array.from(bookingElement.querySelectorAll('button, a')).find(b => ((b.textContent || '').trim().toLowerCase().includes('specify time')));
            if (!specifyBtn) {
                // لا يوجد زر، نتخطى بهدوء
                return false;
            }

            // جلب الوقت من Airtable
            const pickupTime = await fetchAirtablePickupTime(ref);
            if (!pickupTime || !String(pickupTime).trim()) {
                logStep(`No pickup time found in Airtable for ${ref}`, 'WARNING');
                // لا توقف المعالجة إذا لم توجد قيمة في Airtable
                return null;
            }

            // قراءة حالة GYG Pickup Time من Airtable
            let recordId = await getBookingRecordId(ref);
            let gygStatus = null;
            try {
                const fields = await getBookingFromAirtable(recordId);
                const raw = fields && fields["GYG Pickup Time"];
                gygStatus = typeof raw === 'string' ? raw : (raw && raw.name) ? raw.name : null;
            } catch {}

            // سياسة الإرسال: إذا كانت الحالة ليست Done نقوم بالإرسال دائمًا
            const sentKey = `sent_pickup_${ref}`;
            const previouslySent = GM_getValue ? GM_getValue(sentKey, '') : localStorage.getItem(sentKey);
            const isTryAgain = gygStatus && gygStatus.trim().toLowerCase() === 'try again';
            const isDone = gygStatus && gygStatus.trim().toLowerCase() === 'done';
            if (isDone && previouslySent && previouslySent.trim() === String(pickupTime).trim()) {
                logStep(`Pickup time already sent and status Done for ${ref}`, 'INFO');
                return true;
            }
            if (isTryAgain) {
                logStep(`Resending pickup time for ${ref} due to Try Again`, 'INFO');
            } else if (!isDone && previouslySent === String(pickupTime).trim()) {
                logStep(`Forcing resend pickup time for ${ref} to ensure delivery`, 'WARNING');
            }

            // فتح النافذة المنبثقة
            specifyBtn.click();
            await new Promise(r => setTimeout(r, 600));

            // إيجاد عنصر الإدخال داخل الـ dialog (نحدد النافذة المفتوحة فعليًا)
            let dialog = document.querySelector('[role="dialog"], [aria-modal="true"], .ReactModal__Content');
            // انتظر ظهور النافذة لمدة قصيرة إن لم تكن جاهزة فورًا
            const dialogWaitStart = Date.now();
            while (!dialog && Date.now() - dialogWaitStart < 1500) {
                await new Promise(r => setTimeout(r, 100));
                dialog = document.querySelector('[role="dialog"], [aria-modal="true"], .ReactModal__Content');
            }
            dialog = dialog || document;
            // تأكد من التقاط عنصر الإدخال الحقيقي وليس غلاف الـ span (يتشارك نفس الـ id)
            const input = dialog.querySelector('input#datepicker-pickuptime')
                || dialog.querySelector('input[placeholder*="pickup" i]')
                || dialog.querySelector('input[aria-label*="pickup" i]')
                || dialog.querySelector('input.p-datepicker-input')
                || dialog.querySelector('input[role="combobox"][aria-controls]')
                || dialog.querySelector('input[id*="pickuptime" i]')
                || dialog.querySelector('input[type="time"]')
                || dialog.querySelector('input.p-inputtext')
                || dialog.querySelector('input, textarea')
                || dialog.querySelector('[contenteditable="true"]');
            if (!input) {
                logStep('Pickup time input not found in popup', 'ERROR');
                return false;
            }
            // تجهيز صيغ متعددة للوقت لضمان القبول من القالب المستخدم
            function buildTimeCandidates(raw) {
                const s = String(raw || '').trim();
                const m = s.match(/(\d{1,2})[:.](\d{2})\s*(am|pm)?/i);
                if (!m) return [s];
                let h = parseInt(m[1], 10);
                const mm = m[2];
                const ampm = (m[3] || '').toUpperCase();
                let h24 = h;
                if (ampm === 'AM') { h24 = (h === 12) ? 0 : h; }
                else if (ampm === 'PM') { h24 = (h === 12) ? 12 : (h + 12); }
                const hh24 = String(h24).padStart(2, '0');
                const hh12 = String(((h24 % 12) || 12));
                const guessAmPm = ampm || (h24 >= 0 && h24 < 12 ? 'AM' : 'PM');
                return [
                    `${hh24}:${mm}`,           // 24h HH:MM
                    `${h}:${mm}`,              // h:MM 24h (some inputs accept this)
                    `${hh12}:${mm} ${guessAmPm}` // h:MM AM/PM (valid only)
                ];
            }
            const timeCandidates = buildTimeCandidates(pickupTime);

            // قراءة نطاق الوقت الظاهر في النافذة (مثل: between 7:15 AM and 8:00 AM)
            // وإذا كانت قيمة Airtable خارج النطاق، نقوم بتوليد قيمة مقترحة ضمن النطاق لتجربة الإدخال بها أولاً
            try {
                const text = (dialog.innerText || document.body.innerText || '').toUpperCase();
                const mRange = text.match(/BETWEEN\s+(\d{1,2}:\d{2}\s*(AM|PM)?)\s+AND\s+(\d{1,2}:\d{2}\s*(AM|PM)?)/i);
                const toMinutes = (s) => {
                    const m = String(s || '').trim().match(/(\d{1,2}):(\d{2})\s*(AM|PM)?/i);
                    if (!m) return null;
                    let h = parseInt(m[1], 10), mm = parseInt(m[2], 10);
                    const ap = (m[3] || '').toUpperCase();
                    if (ap === 'AM') h = (h === 12) ? 0 : h;
                    else if (ap === 'PM') h = (h === 12) ? 12 : (h + 12);
                    return h * 60 + mm;
                };
                const fmt12 = (minutes) => {
                    minutes = Math.max(0, Math.min(24*60-1, minutes));
                    let h24 = Math.floor(minutes / 60), m = minutes % 60;
                    const ap = h24 >= 12 ? 'PM' : 'AM';
                    let h12 = h24 % 12; h12 = h12 === 0 ? 12 : h12;
                    return `${h12}:${String(m).padStart(2,'0')} ${ap}`;
                };
                if (mRange) {
                    const startStr = mRange[1];
                    const endStr = mRange[3];
                    const sMin = toMinutes(startStr);
                    const eMin = toMinutes(endStr);
                    const pMin = toMinutes(String(pickupTime));
                    if (sMin != null && eMin != null && pMin != null) {
                        // قيد إلى أقرب حد إذا كانت خارج النطاق
                        let clamped = pMin;
                        if (pMin < sMin) clamped = sMin;
                        else if (pMin > eMin) clamped = eMin;
                        // تقريب إلى أقرب 5 دقائق لضمان قبول بعض القوالب
                        clamped = Math.round(clamped / 5) * 5;
                        const suggested = fmt12(clamped);
                        if (!timeCandidates.includes(suggested)) {
                            timeCandidates.unshift(suggested);
                            logStep(`Adjusted pickup time within range: ${suggested}`, 'INFO');
                        }
                    }
                }
            } catch {}

            // حاول أولاً اختيار قيمة من قائمة الاقتراحات (Autocomplete/Listbox) إن وُجدت
            let selectedViaOption = false;
            try {
                input.focus();
                input.click();
                input.dispatchEvent(new KeyboardEvent('keydown', { bubbles: true, key: 'ArrowDown' }));
                input.dispatchEvent(new KeyboardEvent('keyup', { bubbles: true, key: 'ArrowDown' }));
                await new Promise(r => setTimeout(r, 200));
                const normalize = s => String(s || '').replace(/\s+/g,'').toUpperCase();
                const optsSel = '[role="listbox"] [role="option"], .p-autocomplete-items li, .p-listbox-item, .p-dropdown-items li, .p-selectitem';
                let options = Array.from(dialog.querySelectorAll(optsSel));
                const startOptsWait = Date.now();
                while ((!options || options.length === 0) && Date.now() - startOptsWait < 1200) {
                    await new Promise(r => setTimeout(r, 120));
                    // قد تُحقن لوحات الاقتراحات في document.body خارج الـ dialog
                    options = Array.from(document.querySelectorAll(optsSel));
                }
                if (options && options.length > 0) {
                    const candidatesNorm = timeCandidates.map(normalize);
                    const match = options.find(o => {
                        const t = normalize(o.textContent || '');
                        return candidatesNorm.some(c => t.includes(c.replace(/AM|PM/g,'')) || t.includes(c));
                    });
                    if (match) {
                        match.click();
                        input.dispatchEvent(new Event('input', { bubbles: true }));
                        input.dispatchEvent(new Event('change', { bubbles: true }));
                        input.blur();
                        await new Promise(r => setTimeout(r, 200));
                        selectedViaOption = true;
                        logStep(`Selected pickup time via option: ${(match.textContent||'').trim()}`, 'DEBUG');
                    }
                }
            } catch {}

            // ضبط القيمة بشكل متوافق مع مكتبات الواجهات (Prime/React) مع تأكيد التفاعل إذا لم يتم الاختيار عبر عنصر قائمة
            try {
                const desc = Object.getOwnPropertyDescriptor(window.HTMLInputElement.prototype, 'value');
                const valueSetter = desc && desc.set;
                // جرّب كل الصيغ حتى يتم قبولها من الحقل
                for (const candidate of (selectedViaOption ? [] : timeCandidates)) {
                    // تركيز الحقل أولاً لضمان ربط الأحداث
                    input.focus();
                    const isEditable = input.getAttribute && input.getAttribute('contenteditable') === 'true';
                    // إذا كان نوع الحقل time، استخدم صيغة 24h إن أمكن
                    const candidateForType = (input.type === 'time') ? (candidate.match(/^\d{1,2}:\d{2}/) ? candidate.padStart(5, '0') : candidate) : candidate;
                    if (isEditable) {
                        input.textContent = candidateForType;
                    } else {
                        if (valueSetter) {
                            valueSetter.call(input, candidateForType);
                        } else {
                            input.value = candidateForType;
                        }
                        // حافظ على السمة value لبعض القوالب التي تقرأها مباشرة
                        try { input.setAttribute('value', candidateForType); } catch {}
                    }
                    if (input.type === 'time') {
                        // حاول ضبط valueAsDate أيضًا لرفع أحداث التحقق الأصلية
                        const parts = candidateForType.match(/(\d{1,2}):(\d{2})/);
                        if (parts) {
                            const d = new Date();
                            d.setHours(parseInt(parts[1], 10));
                            d.setMinutes(parseInt(parts[2], 10));
                            d.setSeconds(0); d.setMilliseconds(0);
                            try { input.valueAsDate = d; } catch {}
                        }
                    }
                    // تسلسل أحداث لضمان التزامن مع React/Prime
                    input.dispatchEvent(new Event('input', { bubbles: true }));
                    input.dispatchEvent(new KeyboardEvent('keydown', { bubbles: true, key: 'Enter' }));
                    input.dispatchEvent(new Event('change', { bubbles: true }));
                    input.dispatchEvent(new KeyboardEvent('keyup', { bubbles: true, key: 'Enter' }));
                    input.blur();
                    await new Promise(r => setTimeout(r, 200));
                    // إذا لم يعد الحقل فارغًا، نكتفي بهذا المرشح
                    const vNow = isEditable ? (input.textContent || '') : (input.value || '');
                    if (vNow.trim().length >= 4) break;
                    else {
                        logStep(`Typed candidate did not stick: ${candidateForType}`, 'DEBUG');
                    }
                }
            } catch {}
            // تفاعل مباشر مع مُنتقي الوقت إذا كان الحقل يستخدم Time Picker بدلاً من إدخال نصي
            try {
                // افتح اللوحة بالنقر على الحقل ثم انتظر ظهورها
                input.dispatchEvent(new MouseEvent('mousedown', { bubbles: true }));
                input.click();
                await new Promise(r => setTimeout(r, 200));
                // استخدم aria-controls إن وُجد للحصول على لوحة المنتقي المحددة
                const panelId = input.getAttribute('aria-controls');
                let timePanel = (panelId ? document.getElementById(panelId) : null)
                    || dialog.querySelector('.p-timepicker, .p-timepicker-panel, .time-picker, .timepicker, .p-datepicker-time, .p-overlaypanel');
                const panelWaitStart = Date.now();
                while (!timePanel && Date.now() - panelWaitStart < 1200) {
                    await new Promise(r => setTimeout(r, 120));
                    // قد تُحقن اللوحة خارج الـ dialog
                    timePanel = (panelId ? document.getElementById(panelId) : null)
                        || document.querySelector('.p-timepicker, .p-timepicker-panel, .time-picker, .timepicker, .p-datepicker-time, .p-overlaypanel');
                }
                if (timePanel) {

    /**
     * جلب بيانات الحجز من Airtable مع تنظيف الحقول المحسوبة
     */
    function getBookingFromAirtable(recordId) {
        return new Promise((resolve, reject) => {
            if (!recordId) {
                resolve(null);
                return;
            }

            GM_xmlhttpRequest({
                method: "GET",
                url: `https://api.airtable.com/v0/${AIRTABLE_BASE_ID}/${AIRTABLE_TABLE_NAME}/${recordId}`,
                headers: { Authorization: `Bearer ${AIRTABLE_API_KEY}` },
                onload: res => {
                    try {
                        const data = JSON.parse(res.responseText);
                        if (data?.fields) {
                            // تنظيف البيانات المستلمة للاحتفاظ فقط بالحقول المسموح بها
                            const cleanFields = {};
                            ALLOWED_AIRTABLE_FIELDS.forEach(field => {
                                if (data.fields[field] !== undefined) {
                                    cleanFields[field] = data.fields[field];
                                }
                            });
                            resolve(cleanFields);
                        } else {
                            resolve(null);
                        }
                    } catch (e) {
                        reject(e);
                    }
                },
                onerror: reject
            });
        });
    }

    /**
     * حفظ الحجز في Airtable مع تطبيق نهج القائمة البيضاء
     */
    function saveBookingToAirtable(recordId, fields) {
        return new Promise((resolve, reject) => {
            // تطبيق القائمة البيضاء للحقول المسموح بها فقط
            const sanitizedFields = {};
            ALLOWED_AIRTABLE_FIELDS.forEach(field => {
                if (fields[field] !== undefined) {
                    let value = fields[field];
                    // تأكد من أن حقول الأسعار تُرسل كأرقام
                    if (field === "Total price EUR" || field === "Total price USD") {
                        if (typeof value === 'string') {
                            const clean = value.replace(/[^\d.,]/g, '').replace(',', '.');
                            const parsed = parseFloat(clean);
                            value = isNaN(parsed) ? null : parsed;
                        } else if (typeof value !== 'number') {
                            value = null;
                        }
                    }
                    // فرض القيم المسموحة لحقل GYG Pickup Time (Single Select)
                    if (field === "GYG Pickup Time") {
                        const allowed = ["Done", "Try Again"];
                        if (typeof value === 'string') {
                            const trimmed = value.trim();
                            value = allowed.includes(trimmed) ? trimmed : null;
                        } else {
                            value = null;
                        }
                    }
                    // توحيد صيغة التاريخ لحقل GYG Pickup Time Updated At
                    if (field === "GYG Pickup Time Updated At") {
                        try {
                            const d = value ? new Date(value) : new Date();
                            value = d.toISOString();
                        } catch {
                            value = new Date().toISOString();
                        }
                    }
                    sanitizedFields[field] = value;
                }
            });

            GM_xmlhttpRequest({
                method: recordId ? "PATCH" : "POST",
                url: `https://api.airtable.com/v0/${AIRTABLE_BASE_ID}/${AIRTABLE_TABLE_NAME}`,
                headers: {
                    Authorization: `Bearer ${AIRTABLE_API_KEY}`,
                    "Content-Type": "application/json"
                },
                data: JSON.stringify({
                    records: [{ ...(recordId ? { id: recordId } : {}), fields: sanitizedFields }]
                }),
                onload: res => {
                    try {
                        const json = JSON.parse(res.responseText);
                        if (json.error) return reject(json.error);
                        resolve(json);
                    } catch (e) { reject(e); }
                },
                onerror: reject
            });
        });
    }

    function generatePageUrl(p) {
        const { today, nextDate } = getFormattedDate();

    /**
     * التحقق من انتهاء اليوم
     */
    function checkIfDayEnded() {
        const now = new Date(), midnight = new Date();
        midnight.setHours(24, 0, 0, 0);
        if (now >= midnight) {
            logStep("Day ended! Resetting pages...", "WARNING");
            localStorage.setItem("currentPage", 0);
            localStorage.setItem("isNewDay", "true");
            // مسح الذاكرة المؤقتة القديمة عند بداية يوم جديد
            clearOldCache();
            updatePage();
        }
    }

    /**
     * تحديث الصفحة الحالية
     */
    function updatePage() {
        if (localStorage.getItem("isNewDay") === "true") {
            localStorage.setItem("isNewDay", "false");
            currentPage = 0;
        }

        // لا تنتقل للصفحة التالية إذا كان التشغيل متوقفًا
        if (isProcessingPaused) return;

        logStep(`Navigating to page ${currentPage}...`, "PROCESS");
 setTimeout(() => window.location.href = generatePageUrl(currentPage), HUMAN_DELAY.getNextPageDelayMs());
    }

    /**
     * الانتقال إلى الصفحة التالية
     */
    function navigateToNextPage() {
        // لا تنتقل للصفحة التالية إذا كان التشغيل متوقفًا
        if (isProcessingPaused) return;

        // إعادة تعيين عداد الحجوزات المعالجة
        processedBookingsCount = 0;

        currentPage = currentPage >= MAX_PAGES ? 0 : currentPage + 1;
        localStorage.setItem("currentPage", currentPage);
        logStep(`Moving to next page (${currentPage})...`, "PROCESS");
 setTimeout(() => window.location.href = generatePageUrl(currentPage), HUMAN_DELAY.getNextPageDelayMs());
    }

    /**
     * ===================================
     * وظائف معالجة البيانات الأساسية - تم تحسينها مع القائمة البيضاء
     * ===================================
     */

    /**
     * معالجة التغييرات المعتمدة - نسخة محسنة مع القائمة البيضاء
     */
    async function processApprovedChanges(pendingChanges) {
        // التأكد من صحة المدخلات
        if (!Array.isArray(pendingChanges) || pendingChanges.length === 0) {
            logStep("No changes to process", "WARNING");
            isProcessingPaused = false;
            navigateToNextPage();
            return;
        }

        // في الوضع التلقائي، نقوم بتفعيل الموافقة التلقائية بناءً على الإعدادات
        if (AUTO_MODE) {
            pendingChanges.forEach(booking => {
                if (booking.changeType === 'new' && AUTO_SEND_NEW_BOOKINGS && !booking.approved) {
                    booking.approved = true;
                    logStep(`Auto-approved new booking: ${booking["Booking Nr."]}`, "AI");
                } else if (booking.changeType === 'canceled' && AUTO_SEND_CANCELED_BOOKINGS && !booking.approved) {
                    booking.approved = true;
                    logStep(`Auto-approved canceled booking: ${booking["Booking Nr."]}`, "AI");
                } else if (booking.changeType === 'updated' && AUTO_SEND_UPDATES && !booking.approved) {
                    booking.approved = true;
                    logStep(`Auto-approved booking update: ${booking["Booking Nr."]}`, "AI");
                }
            });
        }

        const approvedChanges = pendingChanges.filter(booking => booking.approved);

        if (approvedChanges.length === 0) {
            logStep("No changes approved for processing", "WARNING");
            isProcessingPaused = false;
            navigateToNextPage();
            return;
        }

        logStep(`Processing ${approvedChanges.length} approved changes out of ${pendingChanges.length} total`, "PROCESS");

        const cache = getBookingsCache();
        let successCount = 0;
        let errorCount = 0;

        // تحديث ذاكرة التخزين المؤقت وإرسال البيانات المعتمدة إلى Airtable
        for (const booking of approvedChanges) {
            try {
                const ref = booking["Booking Nr."];
                const recordId = await getBookingRecordId(ref);

                logStep(`${recordId ? "Updating" : "Adding"} booking ${ref} (${booking.changeType})`, recordId ? "INFO" : "INFO");

                // تحسين البيانات باستخدام الذكاء الاصطناعي قبل الإرسال
                const enhancedBooking = await enhanceBookingWithAI(booking);

                // تطبيق القائمة البيضاء على البيانات المحسنة
                const filteredBooking = {};
                ALLOWED_AIRTABLE_FIELDS.forEach(field => {
                    if (enhancedBooking[field] !== undefined) {
                        filteredBooking[field] = enhancedBooking[field];
                    }
                });

                // الحصول على البيانات الموجودة في Airtable إذا كان التحديث
                let finalBookingData = filteredBooking;
                if (recordId) {
                    try {
                        const existingData = await getBookingFromAirtable(recordId);
                        if (existingData) {
                            finalBookingData = mergeBookingData(existingData, filteredBooking);
                            logStep(`Merged data with existing Airtable record for booking ${ref}`, "DEBUG");
                        }
                    } catch (err) {
                        logStep(`Failed to retrieve existing data for ${ref}: ${err.message}`, "WARNING");
                    }
                }

                // إرسال إلى Airtable - باستخدام الحقول المسموحة فقط
                await saveBookingToAirtable(recordId, finalBookingData);

                // تحديث ذاكرة التخزين المؤقت
                cache[ref] = {
                    ...finalBookingData,
                    timestamp: Date.now()
                };

                // تحديث الإحصائيات للنجاح
                successCount++;
                processedBookingsCount++;
                logStep(`Successfully ${recordId ? "updated" : "added"} booking ${ref}`, "SUCCESS", {
                    recordId: recordId,
                    bookingType: booking.changeType
                });

            } catch (err) {
                // تحديث الإحصائيات للأخطاء
                errorCount++;
                logStep(`Failed to process booking ${booking["Booking Nr."]}: ${err.message}`, "ERROR", err);
                logStats.errors++;
                updateLogStats();
            }

            // التحقق من الوصول إلى حد المعالجة
            if (PROCESS_LIMIT > 0 && processedBookingsCount >= PROCESS_LIMIT) {
                logStep(`Reached processing limit of ${PROCESS_LIMIT} bookings`, "WARNING");
                break;
            }
        }

        // حفظ التغييرات في ذاكرة التخزين المؤقت
        saveBookingsCache(cache);

        // عرض ملخص نهائي للعملية
        const summary = `✅ Processed ${successCount} bookings successfully, ${errorCount} failed, out of ${pendingChanges.length} total`;
        logStep(summary, "SUCCESS");

        // تحديث الإحصائيات النهائية
        logStats.processed += successCount;
        updateLogStats();

        // استئناف المعالجة التلقائية
        isProcessingPaused = false;

        // التحقق من الوصول إلى حد المعالجة
        if (PROCESS_LIMIT > 0 && processedBookingsCount >= PROCESS_LIMIT) {
 setTimeout(() => navigateToNextPage(), HUMAN_DELAY.getNextPageDelayMs()); // تأخير بشري عشوائي قبل الانتقال
        } else {
            // استكمال معالجة الصفحة الحالية
            continueProcessing();
        }
    }

    /**
     * استكمال معالجة البيانات بعد الانتهاء من الدفعة الحالية
     */
    function continueProcessing() {
        logStep("Continuing to process remaining bookings...", "PROCESS");
        extractData();
    }

    /**
     * استخراج بيانات الحجوزات من الصفحة
     */
    async function extractData() {
        createLogPanel(); // تأكد من إنشاء لوحة اللوج
        logStep("Starting booking processing cycle...", "PROCESS");

        try {
            logStep("Expanding booking details...", "INFO");
            await expandAllDetails();

            const bookings = document.querySelectorAll('[data-testid="booking-card"]');
            logStep(`Found ${bookings.length} bookings on current page`, "INFO");
            logStats.total = bookings.length;
            updateLogStats();

            if (bookings.length === 0) {
                logStep("No bookings on this page, moving to next page...", "WARNING");
                navigateToNextPage();
                return;
            }

            // حساب عدد الحجوزات المتبقية للمعالجة
            const remainingToProcess = PROCESS_LIMIT > 0 ?
                Math.min(PROCESS_LIMIT - processedBookingsCount, bookings.length) :
                bookings.length;

            logStep(`Processing up to ${remainingToProcess} bookings in this batch`, "PROCESS");

            // استخراج البيانات فقط للعدد المطلوب من الحجوزات
            const bookingsToProcess = Array.from(bookings).slice(0, remainingToProcess);
            const results = [];
            let currentIndex = 0;

            // استخراج بيانات كل حجز مع تحديث شريط التقدم
            for (const booking of bookingsToProcess) {
                currentIndex++;
                const progress = (processedBookingsCount + currentIndex) / PROCESS_LIMIT * 100;
                if (document.getElementById('booking-progress-bar')) {
                    document.getElementById('booking-progress-bar').style.width = `${Math.min(progress, 100)}%`;
                }

                try {
                    const ref = booking.querySelector('.flex:nth-child(2) > p')?.textContent.trim();
                    if (!ref) {
                        logStep(`Skipping booking without reference`, "WARNING");
                        continue;
                    }

                    logStep(`Extracting data for booking: ${ref}`, "DEBUG");

                    // إذا كان هناك إجراء معلق لوقت الالتقاط، حاول الإرسال تلقائياً من Airtable
                    try {
                        const pendingTxt = (booking.innerText || '').toLowerCase();
                        const hasMissingPickup = pendingTxt.includes('pending actions') && pendingTxt.includes('pickup time') && (pendingTxt.includes('missing') || pendingTxt.includes('confirmation required now'));
                        const hasSpecifyBtn = Array.from(booking.querySelectorAll('button,a')).some(b => (b.textContent || '').trim().toLowerCase().includes('specify time'));
                        if (hasMissingPickup || hasSpecifyBtn) {
                            logStep(`Pending pickup time detected for ${ref} — attempting auto fill`, 'PROCESS');
                            const ok = await autoSpecifyPickupTime(booking, ref);
                            if (ok === false) {
                                logStep(`Pausing processing due to pickup send failure for ${ref}`, 'ERROR');
                                isProcessingPaused = true;
                                break;
                            }
                        }
                    } catch (e) {
                        logStep(`Error while trying to auto set pickup for ${ref}: ${e.message}`, 'ERROR');
                    }

                    const productId = booking.querySelector('[data-testid="booking-card-activity-title"]')?.getAttribute('href')?.match(/tour_id=(\d+)/)?.[1] || null;
                    const originalTripName = booking.querySelector('[data-testid="booking-card-activity-title"]')?.textContent.trim() || null;
                    const unifiedTripName = productCodeMap[productId] || originalTripName;
                    const participants = extractParticipants(booking);

                    const bookingStatus = booking.querySelector('[data-testid="booking-status-tag"] .p-tag-label')?.textContent.trim().toLowerCase() || "";
                    let cxlStatus = "Created";
                    if (bookingStatus.includes("canceled")) {
                        cxlStatus = "Canceled";
                        logStep(`Booking ${ref} is canceled`, "BOOKING");
                    } else if (bookingStatus.includes("changed")) {
                        cxlStatus = "Changed";
                        logStep(`Booking ${ref} is changed`, "BOOKING");
                    }

                    // استخراج وتحليل الإضافات باستخدام الذكاء الاصطناعي
                    const addOnsAnalysis = extractAndAnalyzeAddOns(booking);
                    if (addOnsAnalysis.original.length > 0) {
                        logStep(`Found ${addOnsAnalysis.original.length} add-ons for booking ${ref}`, "ADDON");

                        // إذا كانت هناك قيمة للإضافات، نسجلها
                        if (addOnsAnalysis.value > 0) {
                            logStep(`Add-ons value: €${addOnsAnalysis.value.toFixed(2)}`, "ADDON");
                        }

                        // نسجل أنواع الإضافات المكتشفة
                        const categoryTypes = Object.keys(addOnsAnalysis.categories);
                        if (categoryTypes.length > 0) {
                            logStep(`Add-on categories: ${categoryTypes.join(', ')}`, "ADDON");
                        }
                    }

                    // استخراج قيم Retail/Revenue/Commission من قسم breakdown عند توفره
                    const breakdown = await expandBreakdownAndExtract(booking);

                    // تحويل إجمالي السعر باليورو إلى قيمة رقمية دائمًا
                    const totalPriceEURMatch = booking
                        .querySelector('[data-testid="participants-and-price"]')
                        ?.textContent.match(/€\s*([\d.,]+)/);
                    const totalPriceEUR = totalPriceEURMatch
                        ? parseFloat(totalPriceEURMatch[1].replace(',', '.'))
                        : null;

                    const bookingData = {
                        "Agency": "GetYourGuide",
                        "Booking Nr.": ref,
                        "Customer Name": booking.querySelector('[data-testid="lead-traveler-name"]')?.childNodes[0]?.textContent.trim() || null,
                        "Customer Country": booking.querySelector('[data-testid="lead-traveler-name"] span')?.textContent.replace(/[()]/g, '').trim() || null,
                        "Customer Phone": booking.querySelector('[data-testid="lead-traveler-phone"]')?.textContent.trim() || null,
                        "Customer Email": booking.querySelector('[data-testid="lead-traveler-email"]')?.textContent.trim() || null,
                        "trip Name": unifiedTripName,
                        "Product ID": productId,
                        "des": productDestinationMap[productId] || originalTripName?.split(":")[0].trim() || null,
                        "Option": booking.querySelector('p.text-caption.text-label-secondary')?.textContent.replace("Option: ", "").trim() || null,
                        "Date Trip": formatDate(booking.querySelector('p[data-testid="conduction-time"]')?.textContent.trim()),
                        "Total price EUR": totalPriceEUR,
                        // حقول مالية مضافة عبر الدمج
                        "Retail Price": breakdown.retail,
                        "Revenue": breakdown.revenue,
                        "Commission Breakdown": breakdown.commission,
                        "Google Maps": booking.querySelector('[data-testid="custommer-accommodation-google-maps"]')?.getAttribute('href') || null,
                        "Hotel Name": extractHotelName(booking),
                        "Guide": booking.querySelector('[data-testid="booking-detail-conduction-language"] > .text-body')?.textContent.replace("Live guide: ", "").trim() || null,
                        "Traveler name": extractTravelers(booking),
                        "Add - Ons": addOnsAnalysis.formatted || "", // استخدام التنسيق الذكي للإضافات
                        "ADT": participants.Adult,
                        "STD": participants.Student,
                        "CHD": participants.Children,
                        "Inf": participants.Infant,
                        "Youth": participants.Youth,
                        "Booking Status": cxlStatus
                    };

                    results.push(bookingData);

                } catch (err) {
                    logStep(`Error extracting booking data: ${err.message}`, "ERROR", err);
                    logStats.errors++;
                    updateLogStats();
                }
            }

            logStep(`Successfully extracted ${results.length} bookings`, "SUCCESS");

            // اكتشاف التغييرات
            const pendingChanges = detectChanges(results);
            const changeTypes = analyzeChanges(pendingChanges);

            if (pendingChanges.length === 0) {
                logStep("✅ No new changes to process", "SUCCESS");

                // تحقق مما إذا وصلنا إلى حد المعالجة
                if (PROCESS_LIMIT > 0 && processedBookingsCount >= PROCESS_LIMIT) {
                    navigateToNextPage();
                } else if (remainingToProcess < bookings.length) {
                    // استكمال معالجة باقي الحجوزات في الصفحة
                    continueProcessing();
                } else {
                    // انتقل إلى الصفحة التالية إذا تمت معالجة كل الحجوزات
                    navigateToNextPage();
                }
                return;
            }

            // التشغيل التلقائي أو اليدوي
            if (AUTO_MODE) {
                logStep(`Auto mode: processing ${pendingChanges.length} changes...`, "PROCESS");
                processApprovedChanges(pendingChanges);
            } else {
                // عرض واجهة المراجعة اليدوية
                logStep("Showing review UI for manual approval...", "INFO");
                createReviewUI(pendingChanges);
            }

        } catch (err) {
            logStep(`Error processing page: ${err.message}`, "ERROR", err);
 setTimeout(() => navigateToNextPage(), HUMAN_DELAY.getNextPageDelayMs());
        }
    }

    /**
     * تهيئة النظام
     */
    function initializeScript() {
        logStep("Starting GYG Booking Extraction & AI Analysis System v15.3", "INFO");

        // تحقق من إعدادات الوضع التلقائي
        if (AUTO_MODE) {
            logStep("Auto mode enabled: Bookings will be processed without manual intervention", "AI");

            const autoSettings = [];
            if (AUTO_SEND_NEW_BOOKINGS) autoSettings.push("New bookings");
            if (AUTO_SEND_CANCELED_BOOKINGS) autoSettings.push("Canceled bookings");
            if (AUTO_SEND_UPDATES) autoSettings.push("Booking updates");

            logStep(`Auto-processing enabled for: ${autoSettings.join(', ')}`, "AI");
        }

        // تنظيف الذاكرة المؤقتة القديمة
        const removedEntries = clearOldCache();

        // بدء نظام Session Keeper
        startSessionKeeper();
